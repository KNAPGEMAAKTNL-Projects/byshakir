Binding Usage Rules

✓ Always

  • Use #2C1F17 for all body text on light surfaces
  • Use #F0E8DC for all text on dark sections
  • Use #C9A962 as CTA background with #2C1F17 text
  • Use #C9A962 as accent text on dark (#2C1F17) backgrounds
  • Use #8B5230 for terracotta labels, overlines, location text
  • Use #7AABB8 pool blue on dark sections only
  • Keep body background at #F0E8DC or darker — never pure white

✗ Never

  • Use #C9A962 as text color on any light background (1.85:1)
  • Use #F0E8DC light sand as text on gold CTA (1.85:1)
  • Use #7AABB8 on light backgrounds (2.07:1)
  • Use #B87856 for normal-size body text anywhere
  • Use pure white #FFFFFF as body background
  • Use #B87856 warm terracotta for interactive/readable text

/* Generated: 2026-04-07 | Anchor: #C9A962 (logo gold)    */

@theme {
  /* Brand gold — tonal ramp */
  --color-gold:             #C9A962;  /* anchor — CTA bg, decorative */
  --color-gold-light:       #E8C26C;  /* tone-80 — hover on gold */
  --color-gold-dark:        #745B16;  /* tone-40 — pressed/active state */
  --color-gold-subtle:      #FFde86;  /* tone-90 — very light tint */

  /* Terracotta scale */
  --color-terracotta:       #8B5230;  /* AA on bg (5.16:1) — labels, overlines */
  --color-terracotta-warm:  #B87856;  /* Decorative/large text ONLY — card overlays */
  --color-terracotta-deep:  #7A4728;  /* AA on bg (6.26:1) — stronger accent */

  /* Dark surfaces */
  --color-dark:             #2C1F17;  /* espresso — hero, footer, dark sections */
  --color-dark-secondary:   #412E00;  /* warm dark — layering */

  /* Light surfaces */
  --color-bg:               #F0E8DC;  /* body background — warm sand */
  --color-surface:          #E8DDD0;  /* cards, alternating sections */
  --color-surface-elevated: #FAF7F3;  /* hover states, modals */

  /* Text */
  --color-text:             #2C1F17;  /* body text on light (13.14:1) */
  --color-text-muted:       #7A756E;  /* secondary text, captions */
  --color-text-on-dark:     #F0E8DC;  /* text on dark sections (13.14:1) */
  --color-text-muted-dark:  #CEC4B6;  /* muted text on dark */

  /* Borders */
  --color-border:           #CEC4B6;  /* subtle dividers */
  --color-border-strong:    #A89D8D;  /* stronger dividers */

  /* Cool accent — dark bg ONLY */
  --color-cool:             #7AABB8;  /* pool blue — 6.34:1 on dark, FAIL on light */

  /* Semantic */
  --color-error:            #AB0016;
  --color-error-surface:    #FFD8DA;
  --color-success:          #2B4E31;
  --color-success-surface:  #C4EDC8;

  /* Motion */
  --ease-project: cubic-bezier(0.83, 0, 0.17, 1);
}
